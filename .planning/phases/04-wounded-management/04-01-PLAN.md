       public enum WoundedStatus {
           REGISTERED, UNDER_REVIEW, CONFIRMED, NOTIFIED;
           public boolean canTransitionTo(WoundedStatus next) {
               return switch (this) {
                   case REGISTERED -> next == UNDER_REVIEW;
